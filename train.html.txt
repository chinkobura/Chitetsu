async function loadTrain(){

const url="https://buscatch.jp/rt3/unko_map.php?id=chitetsu_train&all=1";

const res=await fetch(url);
const text=await res.text();

const parser=new DOMParser();
const doc=parser.parseFromString(text,"text/html");

const trains=doc.querySelectorAll(".train");

let html="";

trains.forEach(t=>{

const line=t.querySelector(".line")?.innerText || "";
const car=t.querySelector(".car")?.innerText || "";
const dest=t.querySelector(".dest")?.innerText || "";
const speed=t.querySelector(".speed")?.innerText || "";
const pos=t.querySelector(".pos")?.innerText || "";

html+=`
<tr>
<td>${line}</td>
<td>${car}</td>
<td>${dest}</td>
<td>${speed}</td>
<td>${pos}</td>
</tr>
`;

});

document.getElementById("trainTable").innerHTML=html;

}

loadTrain();
setInterval(loadTrain,5000);

</script>
